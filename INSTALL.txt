byTag
function(doc) {   
  if (doc.tagy) {     
	doc.tagy.split(',').forEach(function(tag) {      
		var tTag = tag.trim();      
		if (tTag.length > 0) {        
			emit(tag.trim(), doc);      
		}    
	});  
  }
}
